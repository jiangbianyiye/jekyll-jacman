#### 实现汉诺塔算法，并展示移动过程
```c
#include<stdio.h>
#include<stdlib.h>
#include<malloc.c>

//数组长度，显示用
int N=0;

//显示汉诺塔
void show(int(*a)[3]);

//移动一次，将X杆顶元素移动到Y杆，杆分别用0，1，2代替
void move(int(*a)[3], int n,int X,int Y);

//汉诺塔算法
void hanno(int(*a)[3], int n, int X, int Y, int Z);


int main()
{
	int n = 0;
	scanf("%d", &n);
	N = n;
	int (*a)[3] = calloc(3*n,sizeof(int));
	for (int i = 0; i < n; i++) {
		a[i][0] = i+1;
	}

	show(a);

	hanno(a, n, 0, 1, 2);

	system("pause");
}


void show(int(*a)[3])
{
	printf("\n");
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < 3; j++) {
			if (a[i][j] == 0) {
				printf("%6c", '|');
			}
			else {
				printf("%6d", a[i][j]);
			}
		}
		printf("\n");
	}
	printf("---------------------------\n");
}

void move(int(*a)[3], int n,int X, int Y)
{
	int i = 0, j = 0;
	while (a[i][X] == 0 ) {
		i++;
	}
	while (a[j][Y] == 0) {
		j++;
	}
	int temp = a[i][X];
	a[i][X] = a[j - 1][Y];
	a[j - 1][Y] = temp;

	show(a);// 显示移动过程
}

void  hanno(int(*a)[3], int n, int X, int Y, int Z)
{
	if (n < 1) {
		return;
	}
	else if (n == 1) {
		printf("%c-->%c:\n", 'X' + X, 'X' + Z);
		move(a, n, X, Z);
	}
	else {
		hanno(a, n - 1, X, Z, Y);

		printf("%c-->%c:\n", 'X' + X, 'X' + Z);
		move(a, n, X, Z);

		hanno(a, n - 1, Y, X, Z);
	}
}
```
