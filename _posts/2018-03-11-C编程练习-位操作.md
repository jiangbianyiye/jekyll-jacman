#### 求出一个正整数数的原，反，补码

```c
#include<stdlib.h>
#include<stdio.h>

//原码
void true_code(int num, char* code);
//反码
void complemental_code(int num, char* code);
//补码
void ones_complement_code(int num, char* code);

int main()
{
	int n = 0;
	scanf("%d", &n);

	char *code = calloc(sizeof(int)*8+1,sizeof(char));

	true_code(n,code);
	printf("true code: %s\n", code);

	complemental_code(n, code);
	printf("complemental code: %s\n", code);

	ones_complement_code(n, code);
	printf("ones complement code: %s\n", code);
	system("pause");
}

//原码
void true_code(int num, char* code)
{
	unsigned int sign = 1 << (sizeof(int) * 8 - 1);
	//10000000000
	;
	if (num < 0) {
		num = ~num + 1;	//求补码的原码
		num = num | sign;//处理符号位
	}
	for (int i = 0; i<sizeof(int) * 8; i++) {
		if (num & sign) {
			code[i] = '1';

		}
		else {
			code[i] = '0';
		}
		num <<= 1;
	}
}
//反码
void complemental_code(int num, char* code)
{
	unsigned int sign = 1 << (sizeof(int) * 8 - 1);
	//10000000000
	;
	if (num < 0) {
		num -= 1;
	}
	for (int i = 0; i<sizeof(int) * 8; i++) {
		if (num & sign) {
			code[i] = '1';

		}
		else {
			code[i] = '0';
		}
		num <<= 1;
	}
}
//补码
void ones_complement_code(int num, char* code)
{
	unsigned int sign = 1 << (sizeof(int) * 8 - 1);
	//10000000000
	for (int i = 0; i<sizeof(int) * 8; i++) {
		if (num & sign) {
			code[i] = '1';

		}
		else {
			code[i] = '0';
		}
		num <<= 1;
	}
}

```

#### 用位运算实现整数的加、减、乘、除、取模运算

```c
#include <stdio.h>
#include <stdlib.h>

int add(int a,int b)//加法
{
    int bitsum ;
    int carry ;
    while(b!=0){
         bitsum = a^b; //忽略位进位的位和
         carry = (a&b)<<1; //进位
         a = bitsum;
         b = carry;
    }
    return a;
}

int sub(int a,int b)
{
    add(a,add(~b,1));
}


int multipy(int a,int b)//乘法
{
    //取绝对值
    int absA = a<0?add(~a,1):a;
    int absB = b<0?add(~b,1):b;

    int product = 0;
    while(absB!=0){
        if((absB&1)!=0){
             product = add(absA,product);
        }
         absA<<=1;
         absB>>=1;
    }
    //加上符号
    if((a^b)<0){//符号位
        product = add(~product,1);
    }
    return product;
}

int divide(int a,int b)//求商
{
    //对被除数和除数取绝对值
    int dividend = a<0?add(~a,1):a;
    int divisor = b<0?add(~b,1):b;
    int quotient = 0;
    int remainder = 0;
    //将被除数反序，且在最高位添1，
    //防止在原被除数末尾为多个0时，反序后无法判断位数
    int invert = 2;
    while(dividend)
    {
        invert |= dividend & 1;
        invert <<= 1;
        dividend >>= 1;
    }
    while(invert > 1)//排除最高位的1
    {
        remainder <<= 1;
        remainder |= invert & 1;
        invert >>= 1;
        quotient <<= 1;

        if(remainder >= divisor)
        {
            quotient |= 1;
            remainder = sub(remainder, divisor);
        }
    }
    //求商的符号
    if((a ^ b) < 0)
    {
        quotient = add(~quotient, 1);
    }
    return quotient;
}

int modulo(int a,int b)//求余数
{
    //对被除数和除数取绝对值
    int dividend = a<0?add(~a,1):a;
    int divisor = b<0?add(~b,1):b;
    int quotient = 0;
    int remainder = 0;
    //将被除数反序，且在最高位添1，
    //防止在原被除数末尾为多个0时，反序后无法判断位数
    int invert = 2;
    while(dividend)
    {
        invert |= dividend & 1;
        invert <<= 1;
        dividend >>= 1;
    }
    while(invert > 1)//排除最高位的1
    {
        remainder <<= 1;
        remainder |= invert & 1;
        invert >>= 1;
        quotient <<= 1;

        if(remainder >= divisor)
        {
            quotient |= 1;
            remainder = sub(remainder, divisor);
        }
    }
    //求余数的符号
    if(a < 0)
    {
        remainder= add(~remainder, 1);
    }
    return remainder;
}

int main()
{
    printf("%d,%d,%d\n",add(2,-1),add(-3,4),add(-3,-7));
    printf("%d,%d,%d\n",sub(2,3),sub(5,3),sub(3,-5));
    printf("%d,%d,%d\n",multipy(0,3),multipy(6,3),multipy(3,-5));
    printf("%d,%d,%d\n",divide(9,3),divide(5,2),divide(3,-2));
    printf("%d,%d,%d\n",modulo(9,3),modulo(5,2),modulo(3,-2));
    system("pause");
    return 0;
}
```

#### 将10进制转换为小于等于16的任意进制(考虑任意进制转换)

```c
#include<stdlib.h>
#include<stdio.h>

//将10进制转换位radix进制,radix< =16
void toRadix(int num, char * buffer,int radix);

int main()
{
	int n = 0,r=0;
	while (1) {
		scanf("%d%d", &n, &r);
		char str[sizeof(int) * 8 + 1] = { 0 };
		toRadix(n, str, r);
		printf("%s\n", str);
	}	
}

void toRadix(int num, char * buffer, int radix)
{
	if (radix > 16) {
		return;
	}
	char str[16];
	for (int i = 0; i < 16; i++) {
		if (i < 10) {
			str[i] = '0' + i;
		}
		else {
			str[i] = 'A' + i-10;
		}
	}
	
	char stack[sizeof(int) * 8 + 1] = { 0 };
	int p=0;
	do {
		stack[p++] = str[num % radix];
		num /= radix;
	} while (num);
	p -= 1;
	for (int i=0; p>=0; i++,p--) {
		buffer[i] = stack[p];
	}

}
```
