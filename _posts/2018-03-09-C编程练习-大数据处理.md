#### 硬盘大文件数据检索

###### 单线程硬盘遍历查询

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

void searchStr(char fileToSearch[256],char str[256],char saveFile[256]);

int main()
{
    char str[256]={0};
    scanf("%s",str);//待查询的字符串;
    char fileToSearch[256]="F:\\Class Vedios\\CPP\\大数据相关数据\\开房数据.txt";
    char saveFile[256]={0};
    sprintf(saveFile,"F:\\%s.txt",str);

    time_t start,end;
    time(&start);

    searchStr(fileToSearch,str,saveFile);

    time(&end);
    printf("耗时%f秒\n",difftime(difftime(end,start)));

    system(saveFile);
    system("pause");
    return 0;
}


void searchStr(char fileToSearch[256],char str[256],char saveFile[256])
{
    FILE *pf = fopen(fileToSearch,"r");
    FILE *pfw = fopen(saveFile,"w");

    if(pf==NULL || pfw==NULL){
        printf("can not open file.");
    }
    else{
        char readstr[1024] = {0};
        //feof(pf),表示到达文件尾
        while(!feof(pf)){
            fgets(readstr,1024,pf);//读取一行，缓冲区大小为1024
            char *p = strstr(readstr,str);//查找子字符串
            if(p!=NULL){
                puts(readstr);
                fputs(readstr,pfw);
            }
        }
    }
    fclose(pf);
    fclose(pfw);
    return;
}
```

> 评估：检索平均7~10秒( 平台：i7  5700HQ,7200转机械硬盘)

##### 一级指针内存检索

```c
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<malloc.h>

#define path  "F:\\Class Vedios\\CPP\\大数据相关数据\\八千万QQ密码.txt"

//求文件行数，不考虑处理垃圾数据
int getLineNum(char * filePath)
{
	FILE* pf = fopen(filePath,"r");
	if (pf == NULL)
	{
		printf("文件打开失败");
		return -1;
	}
	int num = 0;
	char buffer[40] = { 0 };
	while(!feof(pf)) {
		fgets(buffer, 40, pf);
		num++;
	}
	fclose(pf);
	return num;
}


//加载文件，构建内存数据
void  loadFile(char * filePath, char *data[]  ) {
	FILE* pf = fopen(filePath, "r");
	if (pf == NULL)
	{
		printf("文件打开失败");
		return ;
	}
	int num = 0;
	char buffer[40] = { 0 };
	while (!feof(pf)) {
		fgets(buffer, 40, pf);
		buffer[39] = '\0';
		int  strlength = strlen(buffer);
		char * entry = calloc(strlength+1, sizeof(char));
		strcpy(entry, buffer);
		data[num++] = entry;
	}
	fclose(pf);
}

//检索数据，输出到命令行
void search(char * filePath, char *data[],char * str)
{
	while (*data != NULL) {
		char *p = strstr(*data, str);
		if (p != NULL)
		{
			puts(*data);//打印
		}
		data++;
	 }
}

int main()
{
	int lineNum = getLineNum(path);
	printf("文件预处理完成\n");
	char ** data = calloc(lineNum+1, sizeof(char *));
	loadFile(path, data);
	printf("数据加载完成\n");
	char str[40] = { 0 };
	while (1) {
		scanf("%s", str);//初始化str数组
		search(path,data,str);//检索
	}

}
```

> 评估：加载到内存10秒以上，检索0~2秒
